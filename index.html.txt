<script>
(() => {
  // --- Config ---
  const MIN = 118.000, MAX = 136.975;
  const STEP_25 = 0.025;
  const STEP_833 = 0.00833; // aproximación suficiente para UI; se cuantiza

  let powerOn = true;
  let use833 = false;

  // Standby is what knobs edit.
  let active = 124.000;
  let standby = 124.850;

  const elActive = document.getElementById('active');
  const elStandby = document.getElementById('standby');
  const pwrBtn = document.getElementById('pwrBtn');
  const swapBtn = document.getElementById('swapBtn');
  const lamp = document.getElementById('lamp');
  const lamp2 = document.getElementById('lamp2');
  const pwrText = document.getElementById('pwrText');
  const stepToggle = document.getElementById('stepToggle');
  const stepMode = document.getElementById('stepMode');
  const stepTxt = document.getElementById('stepTxt');
  const rngA = document.getElementById('rngA');
  const knobMHz = document.getElementById('knobMHz');
  const knobKHz = document.getElementById('knobKHz');

  rngA.textContent = "COM";

  function clamp(x){ return Math.min(MAX, Math.max(MIN, x)); }

  function fmt(f){
    // always show 3 decimals
    return f.toFixed(3);
  }

  function quantize25(f){
    // round to nearest 25 kHz channel
    const k = Math.round((f - Math.floor(f)) / STEP_25);
    const base = Math.floor(f);
    let q = base + k * STEP_25;
    // carry if overflowed decimals
    q = Math.round(q * 1000) / 1000;
    return q;
  }

  function quantize833(f){
    // 8.33 kHz channelization approximation:
    // quantize to 0.005 steps in display is common in some stacks,
    // but for simple sim we quantize to STEP_833 in MHz.
    const base = Math.floor(f);
    const frac = f - base;
    const k = Math.round(frac / STEP_833);
    let q = base + k * STEP_833;
    // snap to 3 decimals display
    q = Math.round(q * 1000) / 1000;
    return q;
  }

  function quantize(f){
    f = clamp(f);
    f = use833 ? quantize833(f) : quantize25(f);
    // after quantization, ensure bounds
    return clamp(f);
  }

  function setStandby(f){
    standby = quantize(f);
    render();
  }

  function setActive(f){
    active = quantize(f);
    render();
  }

  function render(){
    if(powerOn){
      elActive.textContent = fmt(active);
      elStandby.textContent = fmt(standby);
      elActive.style.opacity = "1";
      elStandby.style.opacity = "1";
      lamp.classList.add('on');
      lamp2.classList.remove('on');
      pwrBtn.classList.add('on');
      pwrText.textContent = "ON";
    } else {
      elActive.textContent = "— — — —";
      elStandby.textContent = "— — — —";
      elActive.style.opacity = ".6";
      elStandby.style.opacity = ".6";
      lamp.classList.remove('on');
      lamp2.classList.remove('on');
      pwrBtn.classList.remove('on');
      pwrText.textContent = "OFF";
    }

    stepToggle.classList.toggle('on', use833);
    stepMode.textContent = use833 ? "8.33 kHz" : "25 kHz";
    stepTxt.textContent = use833 ? "8.33" : "25";
  }

  function swap(){
    if(!powerOn) return;
    const tmp = active;
    active = standby;
    standby = tmp;
    render();
  }

  function togglePower(){
    powerOn = !powerOn;
    render();
  }

  function toggleStep(){
    use833 = !use833;
    // re-quantize both to new step so display stays coherent
    active = quantize(active);
    standby = quantize(standby);
    render();
  }

  // --- Knob interaction ---
  function makeKnob(el, onDelta){
    let dragging = false;
    let lastX = 0, lastY = 0;
    let angle = 0;

    const setAngle = (a) => {
      angle = a;
      el.style.transform = `rotate(${angle}deg)`;
    };

    const start = (clientX, clientY) => {
      if(!powerOn) return;
      dragging = true;
      lastX = clientX;
      lastY = clientY;
      el.setPointerCapture?.(event.pointerId);
    };

    const move = (clientX, clientY) => {
      if(!dragging || !powerOn) return;
      const dx = clientX - lastX;
      const dy = clientY - lastY;

      // combine dx and -dy so vertical drag feels natural
      const d = dx + (-dy);

      // speed: 1 step per ~14px
      const steps = Math.trunc(d / 14);

      if(steps !== 0){
        onDelta(steps);
        // rotate for feedback
        setAngle(angle + steps * 6);
        // consume whole steps
        lastX += steps * 14;
        lastY -= steps * 14; // keep drift small
      }
    };

    const end = () => { dragging = false; };

    el.addEventListener('pointerdown', (event) => {
      start(event.clientX, event.clientY);
    });
    el.addEventListener('pointermove', (event) => {
      move(event.clientX, event.clientY);
    });
    el.addEventListener('pointerup', end);
    el.addEventListener('pointercancel', end);

    // mouse wheel: one notch = one step
    el.addEventListener('wheel', (event) => {
      if(!powerOn) return;
      event.preventDefault();
      const dir = Math.sign(event.deltaY); // down = + usually, but invert to feel like "up increases"
      const steps = dir === 0 ? 0 : -dir;
      onDelta(steps);
      setAngle(angle + steps * 6);
    }, {passive:false});
  }

  function stepKHzValue(){
    return use833 ? STEP_833 : STEP_25;
  }

  // MHz knob: step 1.000 MHz
  makeKnob(knobMHz, (steps) => {
    setStandby(standby + steps * 1.000);
  });

  // kHz knob: step channel size
  makeKnob(knobKHz, (steps) => {
    setStandby(standby + steps * stepKHzValue());
  });

  // Buttons
  swapBtn.addEventListener('click', swap);
  pwrBtn.addEventListener('click', togglePower);
  stepToggle.addEventListener('click', toggleStep);

  // Keyboard
  window.addEventListener('keydown', (e) => {
    if(e.repeat) return;
    const k = e.key.toLowerCase();
    if(k === ' '){ e.preventDefault(); swap(); }
    else if(k === 'p'){ togglePower(); }
    else if(k === 'r'){
      powerOn = true; use833 = false;
      active = 124.000; standby = 124.850;
      render();
    }
    else if(e.key === 'ArrowUp'){ if(powerOn) setStandby(standby + stepKHzValue()); }
    else if(e.key === 'ArrowDown'){ if(powerOn) setStandby(standby - stepKHzValue()); }
    else if(e.key === 'ArrowRight'){ if(powerOn) setStandby(standby + 1.0); }
    else if(e.key === 'ArrowLeft'){ if(powerOn) setStandby(standby - 1.0); }
  });

  render();
})();
</script>
